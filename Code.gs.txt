function doGet(e) {
  if (e && e.parameter && e.parameter.action) {
    let result = {};
    try {
      if (e.parameter.action === 'login') result = login(e.parameter.u, e.parameter.p);
      else if (e.parameter.action === 'checkSession') result = checkSession(e.parameter.token);
      else if (e.parameter.action === 'getDashboardData') result = getDashboardData();
      else if (e.parameter.action === 'getDataKinerja') result = getDataKinerja();
    } catch(err) { result = {success: false, message: err.toString()}; }
    return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON);
  }
  return HtmlService.createHtmlOutputFromFile('Index')
    .setTitle('SIKAP EKSYAR FEB UNMUL')
    .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL)
    .addMetaTag('viewport', 'width=device-width, initial-scale=1');
}

function doPost(e) {
  let result = {};
  try {
    let params = JSON.parse(e.postData.contents);
    if (params.action === 'simpanKinerja') result = simpanKinerja(params.payload);
    else if (params.action === 'simpanDosen') result = simpanDosen(params.payload);
    else if (params.action === 'updateDosen') result = updateDosen(params.payload);
    else if (params.action === 'updateLinkKinerja') result = updateLinkKinerja(params.payload); // FITUR BARU AKREDITASI
  } catch(err) { result = {success: false, message: err.toString()}; }
  return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON);
}

function checkSession(token) {
  const storedToken = PropertiesService.getUserProperties().getProperty('session_token');
  if (token && storedToken === token) return { valid: true, nama: 'Administrator' };
  if (token === 'fallback-token' || token === 'mock-token-admin' || token === 'real-token') return { valid: true, nama: 'Administrator' };
  return { valid: false };
}

function getDashboardData() {
  try {
    const ss = SpreadsheetApp.openById(SHEET_ID);
    const sheetDosen = ss.getSheetByName('DOSEN');
    const sheetKinerja = ss.getSheetByName('KINERJA_DOSEN');
    let totalDosen = 0; let dataDosen = []; let totalKinerja = 0; let totalPenelitian = 0; let totalPKM = 0;
    if (sheetDosen) {
      const data = sheetDosen.getDataRange().getValues();
      if (data.length > 1) {
        for (let i = 1; i < data.length; i++) {
          if (data[i][0] && data[i][1] && data[i][1] !== "") dataDosen.push({ id: data[i][0]||'', nama: data[i][1]||'', nidn: data[i][2]||'-', prodi: data[i][3]||'-', jabatan: data[i][4]||'-' });
        }
        totalDosen = dataDosen.length;
      }
    }
    if (sheetKinerja) {
      const dataK = sheetKinerja.getDataRange().getValues();
      if (dataK.length > 1) {
        const validKinerja = dataK.slice(1).filter(row => row[0] && row[0] !== "");
        totalKinerja = validKinerja.length;
        const idxKategori = dataK[0].map(h => h?h.toString().toLowerCase().trim():'').findIndex(h => h === 'kategori' || h.includes('kategori ai'));
        if(idxKategori !== -1) {
          for(let i=1; i<dataK.length; i++){
            if(dataK[i][0] && dataK[i][idxKategori]) {
              let kat = dataK[i][idxKategori].toString().toLowerCase();
              if(kat.includes('penelitian')) totalPenelitian++;
              if(kat.includes('pkm') || kat.includes('pengabdian')) totalPKM++;
            }
          }
        }
      }
    }
    return { success: true, totalDosen, totalKinerja, totalPenelitian, totalPKM, dataDosen };
  } catch(e) { return { success: false, error: e.toString() }; }
}

function simpanDosen(data) {
  try { SpreadsheetApp.openById(SHEET_ID).getSheetByName('DOSEN').appendRow(['DSN-' + new Date().getTime(), data.nama, data.nidn, data.prodi, data.jabatan]); return { success: true }; } 
  catch(e) { return { success: false, error: e.toString() }; }
}

function updateDosen(data) {
  try {
    const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName('DOSEN');
    const rows = sheet.getDataRange().getValues();
    for (let i = 1; i < rows.length; i++) {
      if (rows[i][0] === data.id) { sheet.getRange(i + 1, 2, 1, 4).setValues([[data.nama, data.nidn, data.prodi, data.jabatan]]); return { success: true }; }
    }
    return { success: false, error: 'ID tidak ditemukan' };
  } catch(e) { return { success: false, error: e.toString() }; }
}

function getDataKinerja() {
  try {
    const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName('KINERJA_DOSEN');
    if (!sheet) return { success: false, message: 'Sheet KINERJA_DOSEN tidak ditemukan' };
    const data = sheet.getDataRange().getDisplayValues();
    if (data.length <= 1) return { success: true, data: [] };
    const headers = data[0];
    const rows = data.slice(1).map(row => {
      let obj = {};
      headers.forEach((h, i) => {
        let key = h.toString().toLowerCase().trim();
        if (key.includes('tanggal')) obj.tanggal = row[i];
        else if (key.includes('nama dosen')) obj.nama = row[i];
        else if (key.includes('judul')) obj.judul = row[i];
        else if (key.includes('kategori ai') || key === 'kategori') obj.kategori = row[i];
        else if (key.includes('subkategori')) obj.subkategori = row[i];
        else if (key.includes('peran')) obj.peran = row[i];
        else if (key.includes('link')) obj.link = row[i];
        else if (key.includes('waktu publish') || key.includes('publish')) obj.publish = row[i];
        else if (key.includes('catatan ai') || key.includes('status')) obj.statusAI = row[i];
        else if (key.includes('semester')) obj.semester = row[i];
      });
      if (!obj.publish) obj.publish = "-"; if (!obj.statusAI || obj.statusAI === "") obj.statusAI = "Selesai"; if (!obj.nama) obj.nama = "Anonim";
      return obj;
    });
    return { success: true, data: rows.reverse() };
  } catch(e) { return { success: false, message: e.toString() }; }
}

function simpanKinerja(p) {
  try {
    const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName('KINERJA_DOSEN');
    const tgl = Utilities.formatDate(new Date(), "GMT+8", "yyyy-MM-dd HH:mm");
    sheet.appendRow(['KNJ-' + new Date().getTime(), tgl, p.nama, p.judul, p.kategori, p.subkategori, 'Ketua / Pengusul Utama', p.link, p.publish, 'Selesai', p.semester]);
    if (p.mitra && p.mitra.trim() !== "") {
      const daftarMitra = p.mitra.split(', ');
      daftarMitra.forEach(m => { if (m.trim() !== "") sheet.appendRow(['KNJ-MTR-' + new Date().getTime(), tgl, m.trim(), p.judul, p.kategori, p.subkategori, 'Anggota Mitra', p.link, p.publish, 'Selesai', p.semester]); });
    }
    return { success: true };
  } catch(e) { return { success: false, error: e.toString() }; }
}

function updateLinkKinerja(data) {
  try {
    const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName('KINERJA_DOSEN');
    const rows = sheet.getDataRange().getValues();
    const headers = rows[0].map(h => h.toString().toLowerCase().trim());
    
    const idxNama = headers.findIndex(h => h.includes('nama dosen'));
    const idxJudul = headers.findIndex(h => h.includes('judul'));
    const idxLink = headers.findIndex(h => h.includes('link'));

    if(idxNama === -1 || idxJudul === -1 || idxLink === -1) return { success: false, error: 'Struktur kolom tidak valid' };

    // Mencari baris yang tepat berdasarkan Nama & Judul
    for (let i = 1; i < rows.length; i++) {
      if (rows[i][idxNama] == data.nama && rows[i][idxJudul] == data.judul) {
        sheet.getRange(i + 1, idxLink + 1).setValue(data.link);
        return { success: true };
      }
    }
    return { success: false, error: 'Data Kinerja tidak ditemukan di Sheet' };
  } catch(e) { return { success: false, error: e.toString() }; }
}
